Program to find GCD of two numbers


GCD(Greatest common divisor) of two number is the lagrest number that divides both of them.

60 = 2 X 2 X 3 X 5
36 = 2 X 2 X 3 X 5

GCD = Product of common factors
GCD = 2 X 2 X 3 = 12

The simplest way is to find the prime factors of the two numbers then multipy the common prime factors.

Another way is to apply Euclidean algorithm which is the main algorithm used for this purpose.
The idea is, GCD of two numbers doesn’t change if smaller number is subtracted from a bigger number.
And we are going to use this algorithm.



Using recursion 

    //Harsh Bamotra
    //Program to calculate GCD of two numbers using recursion
                
    #include <iostream>
    using namespace std;
                
    //defining fuction to calculate GCD
    int recur_GCD(int n1 , int n2)
        {
            if(n1==0 || n2==0)     //checking if one of the number is 0
                return 0;
                
            else if(n1==n2)    //if n1 is equal to n2 exit case 
                return n1;
                
            else if(n1>n2)             		//checking if n1 is greater than n2
                return recur_GCD(n1-n2 , n2);
                
            else                          //if n2 is greater than n1
                return recur_GCD(n1 , n2-n1);
        }
                
    int main()
        {
            cout << "***** Calculate GCD of two numbers *****" << endl;
            int n1 , n2;                              //defining variables
            cout << "Enter the first number::";
            cin >> n1;                              //taking first number
            cout << "Enter the second number::";
            cin >> n2;                             //taking second number
                
            cout << "The GCD of the two number::" << recur_GCD(n1 , n2);   //printing the final result
            return 0;
        }

Without using recursion 

    //Harsh Bamotra
    //Program to calculate GCD of two numbers without using recursion
                
    #include <iostream>
    using namespace std;

    // Function to calculate GCD
    int GCD(int x, int y)
        {
	        // Minimum of the two numbers
	        int minimum = min(x, y);

	        if (x % minimum == 0 && y % minimum == 0)
		        return minimum;

	        for (int i = minimum / 2; i >= 2; i--) 
                {
		            if (x % i == 0 && y % i == 0)
			            return i;
	            }
	        return 1;
        }

    int main()
        {
            cout << "***** Calculate GCD of two numbers *****" << endl;
            int n1 , n2;                              //defining variables
            cout << "Enter the first number::";
            cin >> n1;                              //taking first number
            cout << "Enter the second number::";
            cin >> n2;                             //taking second number
                
            cout << "The GCD of the two number::" << GCD(n1 , n2);   //printing the final result
            return 0;
        }
